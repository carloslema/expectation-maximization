  6 //  == User parameters ==
  7 var n = require("numeric");
  8 var gaussian = require('multivariate-gaussian');
  9 
 10 /**
 11  * Fit multivariate data with a gaussian mixture model using the EM algorithm.
 12  * @param {Array<Array<numer>>} points An non-empty array of points. N-dimensional point is represented by an array with N elements.
 13  * @param {number} n_groups number of clusters
 14  * @param {?number} epsilon A parameter indicating when to stop iterating. When the difference in probability varies by less than epsilon between two steps, then consider we have converged.
 15  * @return {Array<{weight:number, mu:Array<number>, sigma:Array<Array<number>>}>} The groups
 16  */
 17 function multivariate_gaussian_fit(points, n_groups, epsilon) {
 18   n_groups = n_groups || 1;
 19   epsilon = epsilon || n.epsilon;
 20 
 21   // dimension of the vector-space of the input data
 22   var dim = points[0].length;
 23 
 24   // == Algorithm ==
 25   // Maximization phase
 26   function compute_groups(tiks) {
 27     var sum = n.sum(tiks);
 28     return tiks.map(function (tik) {
 29       var tiksum = n.sum(tik);
 30       var weight = tiksum / sum;
 31       var mu = n.transpose(n.div(points, tiksum)).map(function (
 32         xs) {
 33         return n.sum(n.mul(xs, tik));
 34       });
 35       var sigma = n.diag(n.rep([dim], n.epsilon));
 36       for (var i = 0; i < points.length; i++) {
 37         var point = points[i];
 38         var diff = [n.sub(point, mu)];
 39         var coeff = tik[i] / tiksum;
 40         var diffdiff = n.dot(n.transpose(diff), diff);
 41         n.addeq(sigma, n.mul(coeff, diffdiff));
 42       }
 43       return {
 44         weight: weight,
 45         sigma: sigma,
 46         mu: mu
 47       };
 48     });
 49   }
 50 
 51   // == Estimation phase ==
 52   function tiks(groups) {
 53     var res = [];
 54     for (var g = 0; g < n_groups; g++) {
 55       var line = [];
 56       var group = groups[g];
 57       // Precompute the density functions
 58       var density_fun = gaussian(group);
 59       for (var p = 0; p < points.length; p++) {
 60         var point = points[p];
 61         line.push(group.weight * density_fun(point));
 62       }
 63       res.push(line);
 64     }
 65     for (var p = 0; p < points.length; p++) {
 66       var sum = 0;
 67       for (var g = 0; g < n_groups; g++) {
 68         sum += res[g][p];
 69       }
 70       for (var g = 0; g < n_groups; g++) {
 71         res[g][p] /= sum;
 72       }
 73     }
 74     return res;
 75   }
 76 
 77   // == Main loop ==
 78   // Attribute points to groups randomly
 79   var ts = n.random([n_groups, points.length]);
 80   // Itereate, alterning estimation and maximization
 81   for (var i = 0; i < 1000; i++) {
 82     var groups = compute_groups(ts);
 83     var oldts = ts;
 84     ts = tiks(groups);
 85     var delta = n.norminf(n.sub(ts, oldts));
 86     if (delta <= epsilon) break;
 87   }
 88   return groups;
 89 }
 90 
 91 module.exports = multivariate_gaussian_fit;
